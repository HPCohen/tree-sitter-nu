def a_command [
     fahren:number # Degrees Fahrenheit
    ] {
    # (100°F − 32) × 5/9 = 37.778°C
    let celcius = $( = $(= $fahren - 32) * 5 / 9)
    build-string $fahren ' °F is ' $celcius ' °C'
}
